# Profit optimization setup
regions = region_month['Region'].unique()

# compute base_sales per region (recent average)
recent = region_month.groupby('Region').tail(6).groupby('Region')['Sales'].mean()
base_sales = recent.to_dict()

# elasticity params (user-tweakable)
alpha = {r: 0.0008 for r in regions}  # effect of marketing spend on sales; tweak per region
beta = 0.8  # diminishing returns

# cost per region: profit = (sales - cost - marketing). We'll use cost = (1 - profit_margin) * sales (or separate)
profit_margin = 0.20

# Objective: maximize total profit = sum( (base_sales + alpha*ms^beta) * margin - ms )
# We'll minimize negative profit.
def total_negative_profit(ms_array, regions_list, base_sales, alpha, beta, margin):
    total = 0.0
    for i, r in enumerate(regions_list):
        ms = ms_array[i]
        sales_new = base_sales[r] + alpha[r] * (ms ** beta)
        profit = sales_new * margin - ms
        total += profit
    return -total  # because we minimize

# constraints: sum(ms) <= Budget, bounds per region
budget = 200000.0  # total budget available (tweakable)
bounds = [(0, budget) for _ in regions]  # you can set region-specific caps

# initial guess: equal allocation
x0 = np.array([budget / len(regions)] * len(regions))

cons = ({'type': 'ineq', 'fun': lambda x: budget - np.sum(x)})  # sum(x) <= budget

res = minimize(total_negative_profit, x0, args=(regions, base_sales, alpha, beta, profit_margin),
               bounds=bounds, constraints=cons, method='SLSQP')

opt_alloc = pd.DataFrame({
    'Region': regions,
    'Optimal_Marketing': res.x
})
opt_alloc['Expected_Profit'] = [ (base_sales[r] + alpha[r] * (res.x[i]**beta)) * profit_margin - res.x[i]
                                for i, r in enumerate(regions) ]

opt_alloc.to_csv('outputs/optimal_allocation.csv', index=False)
print(opt_alloc)
